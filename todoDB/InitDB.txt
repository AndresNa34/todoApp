BEGIN
	CREATE DATABASE TodoDB
END
GO

-- CREATE TASK
IF NOT EXISTS (SELECT * FROM sys.sysobjects WHERE [name] = 'Task' and xtype = 'U')
BEGIN
	CREATE TABLE Task (
		Id INT NOT NULL IDENTITY(1,1),
		Title VARCHAR(50) NOT NULL,
		Completed BIT NOT NULL DEFAULT 0,
		Editing BIT NOT NULL DEFAULT 0,
		CONSTRAINT PKTask PRIMARY KEY (Id)
	)
END
GO
